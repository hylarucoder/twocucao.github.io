3.1 磁盘问题

df -h 查看磁盘块占用的文件（block）
df -i 查看索引节点的占用（Inodes）
find / -size +100M |xargs ls -lh
# 删除 5 天前的文件
find /path/to/files* -mtime +5 -exec rm {} \;
du -h
rm xxx.log
echo "" > xxx.log
